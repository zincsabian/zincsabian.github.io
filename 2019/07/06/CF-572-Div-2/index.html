$$
(a_i^2+a_j^2)(a_i+a_j) \equiv k \pmod{p}\
(a_i^2+a_j^2)(a_i+a_j)(a_i-a_j) \equiv k(a_i-a_j) \pmod{p}\
(a_i^2+a_j^2)(a_i^2-a_j^2)\equiv k(a_i-a_j) \pmod{p}\
a_i^4-a_j^4\equiv ka_i-ka_j \pmod{p}\
a_i^4-ka_i\equiv a_j^4-ka_j
$$
